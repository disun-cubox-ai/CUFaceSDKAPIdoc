341 cu::FaceROIs CUFaceSdkClient::DectectFace(const std::vector<cv::Mat>& frames) {
342  if (frames.size() > 1) {
343  slog::err << "CUFaceSdkClient::DectectFace() Not Implemented!!! frames.size() should be 1 but frames.size() = " << frames.size() << slog::endl;
344  return cu::FaceROIs();
345  }
346  CU_PERF_START("fd");
347  for (size_t i = 0; i < frames.size(); ++i)
348  {
349  auto frame = frames[i];
350  if (frame.empty()) {
351  slog::err << "CUFaceSdkClient::DectectFace() Empty frame!" << slog::endl;
352  return cu::FaceROIs();
353  }
354  CU_Assert(frame.type() == CV_8UC3); // CV_8UC3 = 16, CV_8SC3 = 17
355  CHECK_RESULT(fd_->enqueue(frame.rows, frame.cols, frame.type(), (void*)frame.data));
356  }
357  CHECK_RESULT(fd_->submitRequest());
358 
359  int face_cnt = 0;
360  CHECK_RESULT(fd_->getResultSize(face_cnt));
361 
362  cu::FaceROIs results;
363  for (int i = 0; i < face_cnt; ++i) {
364  cu::Face* face = nullptr;
365  CHECK_RESULT(fd_->getResultAt(i, &face));
366  cv::Rect rect(face->left, face->top, face->width, face->height);
367 
368  cv::Mat input_image = frames[face->batchId];
369  cv::Rect frameRect(0, 0, input_image.cols, input_image.rows);
370  rect = rect & frameRect; // get intersection with frame image
371 
372  cv::Mat face_roi = input_image(rect).clone(); // *** should be copied for align! ***
373 
374  if (ld_)
375  {
376  int size = std::stoi(config_.getValue("lm_image_size", ""));
377  cv::resize(face_roi, face_roi, cv::Size(size, size));
378 
379  if (strcasecmp(config_.getValue("lm_image_layout", "rgb").c_str(), "bgr") == 0)
380  cv::cvtColor(face_roi, face_roi, cv::COLOR_RGB2BGR);
381  }
382 
383  results.push_back(std::make_pair(face, face_roi)); // face_roi.u.refcount 1 -> 2
384  }
385  CU_PERF_END("fd");
386  return results;
387 }

1099  {
1100  return ResultCode::K_ERROR_UNKNOWN;

1115  const auto& face = faces[i].first;
1116

1118  cu::Landmarks landmarks;
1119  if (!face->landmark.landmark.empty() && strcasecmp(client->GetConfigValue("lm_use_fd_landmark").c_str(), "yes") == 0)

1125  retValue.top = face->top;
1126  retValue.width = face->width;
1127  retValue.height = face->height;
1128  retValue.confidence = face->confidence;
1129 
1130  if (face->covid_mask.confidence == 0)
1131  {
1132  cu::CovidMasks mask;
1133  mask = client->DectectCovidMask(roi);
1134 
1135  if(!mask.empty())
1136  retValue.covid_mask_score = mask[0]->confidence;
1137  }
1138  else
1139  {
1140  retValue.covid_mask_score = face->covid_mask.confidence;
1141  }
